import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Game game = new Game();
        game.start();
    }
}

class Game {
    private final Scanner scanner = new Scanner(System.in);
    private final Player p1;
    private final Player p2;
    private final Board board;

    Game() {
        System.out.println("=== ШАШКИ ===");

        System.out.print("Имя игрока 1 (Белые): ");
        String n1 = scanner.nextLine();
        System.out.print("Имя игрока 2 (Чёрные): ");
        String n2 = scanner.nextLine();

        p1 = new Player(n1.isEmpty() ? "Игрок1" : n1, PieceColor.WHITE);
        p2 = new Player(n2.isEmpty() ? "Игрок2" : n2, PieceColor.BLACK);
        board = new Board();
    }

    void start() {
        Player current = p1;

        while (true) {
            clearScreen();
            board.print();

            System.out.println("Ход: " + current.name + " (" + current.color + ")");
            System.out.print("Введите ход (например B6 C5): ");
            String line = scanner.nextLine().toUpperCase();
            String[] parts = line.split(" ");

            if (parts.length != 2) {
                System.out.println("Неверный ввод!");
                pressEnter();
                continue;
            }

            Coordinate from = Coordinate.from(parts[0]);
            Coordinate to = Coordinate.from(parts[1]);
            if (from == null || to == null) {
                System.out.println("Неверные координаты!");
                pressEnter();
                continue;
            }

            if (!board.move(from, to, current.color)) {
                System.out.println("Недопустимый ход!");
                pressEnter();
                continue;
            }

            if (board.checkWin(current.color)) {
                clearScreen();
                board.print();
                System.out.println("ПОБЕДИЛ " + current.name + "!");
                return;
            }

            current = (current == p1) ? p2 : p1;
        }
    }

    private void pressEnter() {
        System.out.println("Нажмите Enter...");
        scanner.nextLine();
    }

    private void clearScreen() {
        for (int i = 0; i < 40; i++) System.out.println();
    }
}

class Player {
    public final String name;
    public final PieceColor color;

    Player(String name, PieceColor color) {
        this.name = name;
        this.color = color;
    }
}

enum PieceColor {WHITE, BLACK}

class Board {
    private final Piece[][] field = new Piece[8][8];

    public Board() {
        for (int y = 0; y < 8; y++) {
            for (int x = 0; x < 8; x++) {
                if ((x + y) % 2 == 1) {
                    if (y < 3) field[y][x] = new Piece(PieceColor.BLACK);
                    if (y > 4) field[y][x] = new Piece(PieceColor.WHITE);
                }
            }
        }
    }

    public boolean move(Coordinate from, Coordinate to, PieceColor color) {
        Piece p = field[from.y][from.x];
        if (p == null || p.color != color) return false;
        if (field[to.y][to.x] != null) return false;

        int dx = to.x - from.x;
        int dy = to.y - from.y;
        int dir = (p.color == PieceColor.WHITE) ? -1 : 1;

        if (!p.isKing) {
            // обычный ход
            if (Math.abs(dx) == 1 && dy == dir) {
                field[to.y][to.x] = p;
                field[from.y][from.x] = null;
                makeKing(to);
                return true;
            }

            // взятие
            if (Math.abs(dx) == 2 && Math.abs(dy) == 2) {
                int mx = from.x + dx / 2;
                int my = from.y + dy / 2;
                Piece enemy = field[my][mx];
                if (enemy != null && enemy.color != p.color) {
                    field[my][mx] = null;
                    field[to.y][to.x] = p;
                    field[from.y][from.x] = null;
                    makeKing(to);
                    return true;
                }
            }
        } else {
            if (Math.abs(dx) == Math.abs(dy)) {
                int x_dir = (to.x - from.x) / Math.abs(dx);
                int y_dir = (to.y - from.y) / Math.abs(dy);

                int p_count = 0;

                int ex = 0;
                int ey = 0;

                for (int i = 1; i <= Math.abs(dx); i++) {
                    int x = from.x + i * x_dir;
                    int y = from.y + i * y_dir;
                    if (field[y][x] != null) {
                        p_count++;
                        ex = x;
                        ey = y;
                    }
                }

                if (p_count > 1) return false;
                else if (p_count == 1) {
                    field[ey][ex] = null;
                    field[to.y][to.x] = p;
                    field[from.y][from.x] = null;
                    return true;
                } else {
                    field[to.y][to.x] = p;
                    field[from.y][from.x] = null;
                    return true;
                }
            }
        }
        return false;
    }

    private void makeKing(Coordinate c) {
        Piece p = field[c.y][c.x];
        if (p.color == PieceColor.WHITE && c.y == 0) p.isKing = true;
        if (p.color == PieceColor.BLACK && c.y == 7) p.isKing = true;
    }

    public boolean checkWin(PieceColor color) {
        PieceColor enemy_color;

        if (color == PieceColor.WHITE) {
            enemy_color = PieceColor.BLACK;
        } else {
            enemy_color = PieceColor.WHITE;
        }

        for (int y = 0; y < 8; y++) {
            for (int x = 0; x < 8; x++) {
                if (field[y][x] != null) {
                    PieceColor piece_color = field[y][x].color;
                    if (piece_color == enemy_color) {
                        return false;
                    }
                }
            }
        }
        return true;
    }

    public void print() {
        System.out.println("  A B C D E F G H");
        for (int y = 0; y < 8; y++) {
            System.out.print((8 - y) + " ");
            for (int x = 0; x < 8; x++) {
                Piece p = field[y][x];
                if (p == null) System.out.print(". ");
                else if (p.color == PieceColor.WHITE)
                    System.out.print(p.isKing ? "W " : "w ");
                else
                    System.out.print(p.isKing ? "B " : "b ");
            }
            System.out.println();
        }
    }
}

class Piece {
    public final PieceColor color;
    public boolean isKing;

    public Piece(PieceColor color) {
        this.color = color;
        this.isKing = false;
    }
}

class Coordinate {
    public int x, y;

    public Coordinate(int x, int y) {
        this.x = x;
        this.y = y;
    }

    public static Coordinate from(String s) {
        if (s.length() != 2) return null;
        char c = s.charAt(0);
        char n = s.charAt(1);
        int x = c - 'A';
        int y = 8 - (n - '0');
        if (x < 0 || x > 7 || y < 0 || y > 7) return null;
        return new Coordinate(x, y);
    }
}
