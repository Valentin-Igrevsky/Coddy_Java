import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Game game = new Game();
        game.start();
    }
}

class Game {
    private final Scanner scanner = new Scanner(System.in);
    private final Player p1;
    private final Player p2;
    private final Board board;

    Game() {
        System.out.println("=== Четыре в ряд ===");

        System.out.print("Имя игрока 1 (X): ");
        String n1 = scanner.nextLine();
        System.out.print("Имя игрока 2 (O): ");
        String n2 = scanner.nextLine();

        p1 = new Player(n1.isEmpty() ? "Игрок1" : n1, 'X');
        p2 = new Player(n2.isEmpty() ? "Игрок2" : n2, 'O');

        board = new Board();
    }

    void start() {
        Player current = p1;

        while (true) {
            clear();
            board.print();
            System.out.println("Ход: " + current.name + " (" + current.symbol + ")");

            System.out.print("Введите номер столбца (1–7): ");
            String input = scanner.nextLine();

            int col;
            try {
                col = Integer.parseInt(input) - 1;
            } catch (Exception e) {
                System.out.println("Неверный ввод!");
                pause();
                continue;
            }

            if (!board.drop(col, current.symbol)) {
                System.out.println("Нельзя поставить сюда!");
                pause();
                continue;
            }

            if (board.hasWinner(current.symbol)) {
                clear();
                board.print();
                System.out.println("ПОБЕДА! Выиграл " + current.name + "!");
                return;
            }

            if (board.isFull()) {
                clear();
                board.print();
                System.out.println("НИЧЬЯ! Поле заполнено.");
                return;
            }

            current = (current == p1) ? p2 : p1;
        }
    }

    private void pause() {
        System.out.println("Нажмите Enter...");
        scanner.nextLine();
    }

    private void clear() {
        for (int i = 0; i < 40; i++) System.out.println();
    }
}

class Player {
    public final String name;
    public final char symbol;

    Player(String name, char symbol) {
        this.name = name;
        this.symbol = symbol;
    }
}

class Board {
    private static final int ROWS = 6;
    private static final int COLS = 7;

    private final char[][] grid = new char[ROWS][COLS];

    public Board() {
        for (int y = 0; y < ROWS; y++)
            for (int x = 0; x < COLS; x++)
                grid[y][x] = '.';
    }

    public boolean drop(int col, char symbol) {
        if (col < 0 || col >= COLS) return false;

        for (int row = ROWS - 1; row >= 0; row--) {
            if (grid[row][col] == '.') {
                grid[row][col] = symbol;
                return true;
            }
        }
        return false;
    }

    public boolean isFull() {
        for (int x = 0; x < COLS; x++)
            if (grid[0][x] == '.') return false;
        return true;
    }

    public boolean hasWinner(char s) {
        // горизонталь
        for (int y = 0; y < ROWS; y++)
            for (int x = 0; x <= COLS - 4; x++)
                if (grid[y][x] == s && grid[y][x+1] == s && grid[y][x+2] == s && grid[y][x+3] == s)
                    return true;

        // вертикаль
        for (int x = 0; x < COLS; x++)
            for (int y = 0; y <= ROWS - 4; y++)
                if (grid[y][x] == s && grid[y+1][x] == s && grid[y+2][x] == s && grid[y+3][x] == s)
                    return true;

        // диагональ ↘
        for (int y = 0; y <= ROWS - 4; y++)
            for (int x = 0; x <= COLS - 4; x++)
                if (grid[y][x] == s && grid[y+1][x+1] == s && grid[y+2][x+2] == s && grid[y+3][x+3] == s)
                    return true;

        // диагональ ↙
        for (int y = 0; y <= ROWS - 4; y++)
            for (int x = 3; x < COLS; x++)
                if (grid[y][x] == s && grid[y+1][x-1] == s && grid[y+2][x-2] == s && grid[y+3][x-3] == s)
                    return true;

        return false;
    }

    public void print() {
        System.out.println("  1 2 3 4 5 6 7");
        for (int y = 0; y < ROWS; y++) {
            System.out.print((y + 1) + " ");
            for (int x = 0; x < COLS; x++) {
                System.out.print(grid[y][x] + " ");
            }
            System.out.println();
        }
    }
}
