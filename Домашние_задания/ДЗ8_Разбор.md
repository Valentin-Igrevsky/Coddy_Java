public class Main {
    public static void main(String[] args) {
        Player player = new Player("Valentin");
        Sword enemy_sword = new Sword(12, 50);
        Sword enemy_sword_2 = new Sword(12, 60);

        Medkit player_medkit = new Medkit(10, 5);


        int given_damage = player.damage(enemy_sword);
        System.out.printf("-%s\n", given_damage);
        enemy_sword.stat();
        enemy_sword_2.stat();

        int given_heal = player.eat_medkit(player_medkit);
        System.out.printf("+%s\n", given_heal);
        player_medkit.stat();

        player.stat();
    }
}

class Player {
    private String name;
    private int hp;
    private final int max_hp = 100;

    Player(String name) {
        this.hp = 100;
        this.name = name;
    }

    public int damage(Sword sword) {
        int sword_damage = sword.hit();
        int remaining_health = this.hp - sword_damage;
        if (remaining_health <= 0) {
            int taken_damage = this.hp;
            this.hp = 0;
            return taken_damage;
        } else {
            this.hp -= sword_damage;
            return sword_damage;
        }
    }

    private int heal(int heal_hp) {
        int new_hp = this.hp + heal_hp;
        if (new_hp > max_hp) {
            int taken_hp = heal_hp - (new_hp - this.max_hp);
            this.hp = max_hp;
            return taken_hp;
        } else {
            this.hp -= heal_hp;
            return heal_hp;
        }
    }

    public int eat_medkit(Medkit medkit) {
        return heal(medkit.use());
    }

    public void stat() {
        System.out.println("name: " + name);
        System.out.println("hp " + hp);
    }
}

class Sword {
    private int base_damage;
    private int durability;

    Sword(int base_damage, int durability) {
        this.base_damage = base_damage;
        this.durability = durability;
    }

    public int hit() {
        if (this.durability > 0) {
            this.durability--;
            return this.base_damage;
        } else {
            return 0;
        }
    }

    public void stat() {
        System.out.println("base_damage: " + base_damage);
        System.out.println("durability: " + durability);
    }
}

class Medkit {
    private int heal_points;
    private int count;

    Medkit(int heal_points, int count) {
        this.heal_points = heal_points;
        this.count = count;
    }

    public int use() {
        if (this.count > 0) {
            this.count--;
            return this.heal_points;
        } else {
            return 0;
        }
    }

    public void stat() {
        System.out.println("heal_points: " + heal_points);
        System.out.println("count: " + count);
    }
}
